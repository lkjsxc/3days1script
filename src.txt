fn print_int ( c ) (
    if c == 0 (
        write ( 1 48 )
        write ( 1 10 )
        return 0
    )
    if c < 0 (
        write ( 1 45 )
        &c = - c
    )
    write ( 1 c + 48 )
    write ( 1 10 )
    return 0
)

fn mul ( a b ) (
    return a * b
)

&a = print_int ( 0 )
&a = print_int ( 1 )
&a = print_int ( -1 )
